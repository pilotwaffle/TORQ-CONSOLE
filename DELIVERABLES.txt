USAGE EXAMPLES

Example 1: Basic Progress Tracking
  from torq_console.llm.providers.progress import ProgressTracker

  tracker.complete()

Example 2: Multi-Format Export
  from torq_console.llm.providers.export import ExportManager

  exporter = ExportManager()
  markdown = exporter.export_to_markdown(results)
  pdf_path = exporter.export_to_pdf(results, "output.pdf")
  csv_path = exporter.export_to_csv(results, "output.csv")

Example 3: Research with Progress Tracking
  from torq_console.llm.providers.websearch import WebSearchProvider

  provider = WebSearchProvider()
  results = await provider.research_topic_with_progress(
      topic="AI Trends",
      depth="deep",
      progress_callback=lambda s: print(f"{s['percent']:.0f}%")
  )
